#include <stdio.h>
#include <string.h>

#define MAX 100

int stack[MAX];
int top = -1;

void push(int x) {
    stack[++top] = x;
}

int pop() {
    if (top == -1) return -1;
    return stack[top--];
}

int esteValid(char* s) {
    for (int i = 0; s[i]; i++) {
        if (s[i] == '(')
            push('(');
        else if (s[i] == ')') {
            if (pop() == -1) return 0;
        }
    }
    return top == -1;
}

int main() {
    printf("%d", esteValid("()(())"));
    return 0;
}
