#include <stdio.h>

void selectionSort(int v[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < n; j++)
            if (v[j] < v[minIdx])
                minIdx = j;
        int temp = v[i];
        v[i] = v[minIdx];
        v[minIdx] = temp;
    }
}

int main() {
    int v[] = {64, 25, 12, 22, 11};
    int n = 5;
    selectionSort(v, n);
    for (int i = 0; i < n; i++)
        printf("%d ", v[i]);
    return 0;
}
