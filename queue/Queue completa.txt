#include <stdio.h>
#define MAX 100

int queue[MAX];
int front = 0, rear = -1;

void enqueue(int x) {
    if (rear < MAX - 1)
        queue[++rear] = x;
}

int dequeue() {
    if (front > rear) return -1;
    return queue[front++];
}

int isEmpty() {
    return front > rear;
}

int main() {
    enqueue(1);
    enqueue(2);
    enqueue(3);
    while (!isEmpty())
        printf("%d ", dequeue());
    return 0;
}
