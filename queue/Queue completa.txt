#include <stdio.h>
#define MAX 100

int queue[MAX];
int front = 0, rear = -1;

int isEmpty() {
    return front > rear;
}

void enqueue(int x) {
    if (rear == MAX - 1) return;
    queue[++rear] = x;
}

int dequeue() {
    if (isEmpty()) return -1;
    return queue[front++];
}

int main() {
    enqueue(10);
    enqueue(20);
    enqueue(30);
    printf("%d ", dequeue());
    printf("%d ", dequeue());
    enqueue(40);
    printf("%d", dequeue());
    return 0;
}
