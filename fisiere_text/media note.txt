#include <stdio.h>

int main() {
    FILE *f = fopen("date.txt", "r");
    char nume[50];
    int nota, count = 0, suma = 0;

    while (fscanf(f, "%s %d", nume, &nota) != EOF) {
        suma += nota;
        count++;
    }

    fclose(f);
    if (count > 0)
        printf("%.2f", (float)suma / count);
    return 0;
}
