#include <stdio.h>
#include <stdlib.h>

typedef struct Nod {
    int data;
    struct Nod* next;
    struct Nod* prev;
} Nod;

Nod* inserareSfarsit(Nod* head, int val) {
    Nod* nou = (Nod*)malloc(sizeof(Nod));
    nou->data = val;
    nou->next = NULL;
    nou->prev = NULL;

    if (head == NULL) return nou;

    Nod* temp = head;
    while (temp->next != NULL)
        temp = temp->next;

    temp->next = nou;
    nou->prev = temp;
    return head;
}

void afisare(Nod* head) {
    while (head != NULL) {
        printf("%d ", head->data);
        head = head->next;
    }
}

int main() {
    Nod* lista = NULL;
    lista = inserareSfarsit(lista, 1);
    lista = inserareSfarsit(lista, 2);
    lista = inserareSfarsit(lista, 3);
    afisare(lista);
    return 0;
}
